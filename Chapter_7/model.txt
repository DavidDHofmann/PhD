model {

  # Priors
  beta0 ~ dnorm(0, 0.001) # Intercept
  beta1 ~ dnorm(0, 0.001) # Beta for maxMoonlightIntensity
  beta2 ~ dnorm(0, 0.001) # Beta for maxTemperature
  beta3 ~ dnorm(0, 0.001) # Beta for maxTemperature:maxMoonlightIntensity
  sd    ~ dunif(0, 100)
  variance  <- sd * sd
  precision <- 1 / variance

  # Likelihood
  for (i in 1:n) {
    meanActX[i] ~ dnorm(mean[i], precision)
    mean[i] <- beta0 +
      beta1 * maxMoonlightIntensity[i] +
      beta2 * maxTemperature[i] +
      beta3 * maxTemperature[i] * maxMoonlightIntensity[i]
  }

}
